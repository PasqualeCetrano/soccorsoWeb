DROP DATABASE soccorso;
CREATE DATABASE soccorso; 
USE soccorso; 
  
CREATE TABLE Patente ( 
    id_patente INT PRIMARY KEY AUTO_INCREMENT, 
    codice_patente VARCHAR(10) NOT NULL UNIQUE, 
    tipo VARCHAR(50) NOT NULL 
); 
  
CREATE TABLE Abilita ( 
    id_abilita INT PRIMARY KEY AUTO_INCREMENT, 
    descrizione VARCHAR(100) NOT NULL UNIQUE  
); 
  
CREATE TABLE Utente ( 
    id_utente INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    indirizzo VARCHAR(100), 
    tipo ENUM('amministratore', 'operatore') NOT NULL, 
    nascita DATE, 
    email VARCHAR(100) UNIQUE NOT NULL,
    telefono VARCHAR(20) NOT NULL UNIQUE,
    nome VARCHAR(50) NOT NULL, 
    cognome VARCHAR(50) NOT NULL, 
    codicefiscale VARCHAR(16) UNIQUE, 
    password VARCHAR(255) NOT NULL, 
    id_utente_amministratore INT, 
    FOREIGN KEY (id_utente_amministratore) REFERENCES Utente(id_utente) ON DELETE SET 
NULL 
); 
  
  
  
CREATE TABLE Richiesta_Soccorso ( 
    id_richiesta_soccorso INT PRIMARY KEY AUTO_INCREMENT, 
    stato ENUM('attiva', 'in_corso', 'chiuso','annullata','da convalidare') NOT NULL, 
    foto LONGBLOB, 
    stringa_convalida VARCHAR(255) DEFAULT NULL UNIQUE, 
    ip VARCHAR(45) NOT NULL, 
    descrizione TEXT NOT NULL, 
    coordinate VARCHAR(100) NOT NULL, 
    ora_invio DATETIME NOT NULL, 
    email_segnalante VARCHAR(100) NOT NULL, 
    segnalante VARCHAR(50) NOT NULL   
); 
  
CREATE TABLE Missione ( 
    id_missione INT PRIMARY KEY AUTO_INCREMENT, 
    posizione VARCHAR(100) NOT NULL, 
    obiettivo TEXT NOT NULL, 
    livello_successo TINYINT CHECK (livello_successo BETWEEN 0 AND 5), 
    commenti TEXT, 
    fine DATETIME, 
    inizio DATETIME NOT NULL, 
    fk_id_utente INT DEFAULT NULL, 
    fk_id_richiesta_soccorso INT NOT NULL, 
    FOREIGN KEY (fk_id_utente) REFERENCES Utente(id_utente), 
    FOREIGN KEY (fk_id_richiesta_soccorso) REFERENCES 
Richiesta_Soccorso(id_richiesta_soccorso) 
); 
  
CREATE TABLE Squadra ( 
    id_squadra INT PRIMARY KEY AUTO_INCREMENT, 
    codice VARCHAR(50) UNIQUE NOT NULL, 
    fk_id_missione INT NOT NULL, 
    FOREIGN KEY (fk_id_missione) REFERENCES Missione(id_missione) 
); 
  
CREATE TABLE Aggiornamento ( 
    id_aggiornamento INT PRIMARY KEY AUTO_INCREMENT, 
    timestamp_agg DATETIME NOT NULL, 
    testo TEXT NOT NULL, 
    fk_id_missione INT NOT NULL, 
    fk_id_utente INT NOT NULL, 
    FOREIGN KEY (fk_id_missione) REFERENCES Missione(id_missione), 
    FOREIGN KEY (fk_id_utente) REFERENCES Utente(id_utente) 
); 
  
  
CREATE TABLE Mezzo ( 
    id_mezzo INT PRIMARY KEY AUTO_INCREMENT, 
    nome VARCHAR(50) NOT NULL UNIQUE , 
    descrizione TEXT 
); 
  
CREATE TABLE Materiale ( 
    id_materiale INT PRIMARY KEY AUTO_INCREMENT, 
    nome VARCHAR(50) NOT NULL UNIQUE , 
    descrizione TEXT 
); 
  
CREATE TABLE Partecipa ( 
    id_partecipa INT PRIMARY KEY AUTO_INCREMENT, 
    fk_id_squadra INT NOT NULL, 
    fk_id_utente INT NOT NULL, 
    ruolo ENUM('caposquadra', 'operatore') NOT NULL, 
    FOREIGN KEY (fk_id_squadra) REFERENCES Squadra(id_squadra), 
    FOREIGN KEY (fk_id_utente) REFERENCES Utente(id_utente) 
); 
  
CREATE TABLE Impiega_Mezzo ( 
    id_impiega_mezzo INT PRIMARY KEY AUTO_INCREMENT, 
    fk_id_mezzo INT NOT NULL, 
    fk_id_missione INT NOT NULL, 
    FOREIGN KEY (fk_id_mezzo) REFERENCES Mezzo(id_mezzo), 
    FOREIGN KEY (fk_id_missione) REFERENCES Missione(id_missione) 
); 
  
CREATE TABLE Impiega_Materiale ( 
    id_impiega_materiale INT PRIMARY KEY AUTO_INCREMENT, 
    fk_id_materiale INT NOT NULL, 
    fk_id_missione INT NOT NULL, 
    FOREIGN KEY (fk_id_materiale) REFERENCES Materiale(id_materiale), 
    FOREIGN KEY (fk_id_missione) REFERENCES Missione(id_missione) 
); 
  
CREATE TABLE Detiene ( 
    id_detiene INT PRIMARY KEY AUTO_INCREMENT, 
    fk_id_patente INT NOT NULL, 
    fk_id_utente INT NOT NULL, 
    FOREIGN KEY (fk_id_patente) REFERENCES Patente(id_patente), 
    FOREIGN KEY (fk_id_utente) REFERENCES Utente(id_utente) 
); 
  
CREATE TABLE Possiede ( 
    id_possiede INT PRIMARY KEY AUTO_INCREMENT, 
    fk_id_abilita INT NOT NULL, 
    fk_id_utente INT NOT NULL, 
    FOREIGN KEY (fk_id_abilita) REFERENCES Abilita(id_abilita), 
    FOREIGN KEY (fk_id_utente) REFERENCES Utente(id_utente) 
);
